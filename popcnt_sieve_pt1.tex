%     qubit   \tilde{v}_{1}
%     qubit   a1,0
%     qubit   \tilde{u}_{1}
%     qubit   \tilde{u}_{2}
%     qubit   a2,0
%     qubit   \tilde{v}_{2}
%     qubit   \tilde{v}_{3}
%     qubit   a3,0
%     qubit   \tilde{u}_{3}
% 
%     cnot    \tilde{u}_{0},\tilde{v}_{0}
%     cnot    \tilde{u}_{1},\tilde{v}_{1}
%     cnot    \tilde{u}_{2},\tilde{v}_{2}
%     cnot    \tilde{u}_{3},\tilde{v}_{3}
% 
%     add4    a0,\tilde{v}_{0},\tilde{v}_{1},a1
%     add4    a2,\tilde{v}_{2},\tilde{v}_{3},a3

%  Time 01:
%    Gate 00 cnot(\tilde{u}_{0},\tilde{v}_{0})
%    Gate 01 cnot(\tilde{u}_{1},\tilde{v}_{1})
%    Gate 02 cnot(\tilde{u}_{2},\tilde{v}_{2})
%    Gate 03 cnot(\tilde{u}_{3},\tilde{v}_{3})
%  Time 02:
%    Gate 04 add4(a0,\tilde{v}_{0},\tilde{v}_{1},a1)
%    Gate 05 add4(a2,\tilde{v}_{2},\tilde{v}_{3},a3)

% Qubit circuit matrix:
%
% \tilde{u}_{0}: gAxA, n  , n  
% a0: n  , gBxB, n  
% \tilde{v}_{0}: gAxC, gBxC, n  
% \tilde{v}_{1}: gAxD, gBxD, n  
% a1: n  , gBxE, n  
% \tilde{u}_{1}: gAxF, n  , n  
% \tilde{u}_{2}: gAxG, n  , n  
% a2: n  , gBxH, n  
% \tilde{v}_{2}: gAxI, gBxI, n  
% \tilde{v}_{3}: gAxJ, gBxJ, n  
% a3: n  , gBxK, n  
% \tilde{u}_{3}: gAxL, n  , n  

\documentclass[11pt]{article}
\input{xyqcirc.tex}

% definitions for the circuit elements

\def\gAxA{\b\w\A{gAxA}}
\def\gAxC{\o\w\A{gAxC}}
\def\gAxF{\b\w\A{gAxF}}
\def\gAxD{\o\w\A{gAxD}}
\def\gAxG{\b\w\A{gAxG}}
\def\gAxI{\o\w\A{gAxI}}
\def\gAxL{\b\w\A{gAxL}}
\def\gAxJ{\o\w\A{gAxJ}}
\def\gBxB{\gnqubit{\texttt{add}}{ddd}\w\A{gBxB}}
\def\gBxC{\gspace{\texttt{add}}\w\A{gBxC}}
\def\gBxD{\gspace{\texttt{add}}\w\A{gBxD}}
\def\gBxE{\gspace{\texttt{add}}\w\A{gBxE}}
\def\gBxH{\gnqubit{\texttt{add}}{ddd}\w\A{gBxH}}
\def\gBxI{\gspace{\texttt{add}}\w\A{gBxI}}
\def\gBxJ{\gspace{\texttt{add}}\w\A{gBxJ}}
\def\gBxK{\gspace{\texttt{add}}\w\A{gBxK}}

% definitions for bit labels and initial states

\def\bA{ \q{\tilde{u}_{0}}}
\def\bB{\qv{a_{0}}{0}}
\def\bC{ \q{\tilde{v}_{0}}}
\def\bD{ \q{\tilde{v}_{1}}}
\def\bE{\qv{a_{1}}{0}}
\def\bF{ \q{\tilde{u}_{1}}}
\def\bG{ \q{\tilde{u}_{2}}}
\def\bH{\qv{a_{2}}{0}}
\def\bI{ \q{\tilde{v}_{2}}}
\def\bJ{ \q{\tilde{v}_{3}}}
\def\bK{\qv{a_{3}}{0}}
\def\bL{ \q{\tilde{u}_{3}}}

% The quantum circuit as an xymatrix

\xymatrix@R=5pt@C=10pt{

\\  \bD & \gAxD &\gBxD &\n  
\\  \bE & \n   &\gBxE &\n  
\\  \bF & \gAxF &\n   &\n  
\\  \bG & \gAxG &\n   &\n  
\\  \bH & \n   &\gBxH &\n  
\\  \bI & \gAxI &\gBxI &\n  
\\  \bJ & \gAxJ &\gBxJ &\n  
\\  \bK & \n   &\gBxK &\n  
\\  \bL & \gAxL &\n   &\n  
%
